/* Develop a Java-based library management application that demonstrates OOP
principles and uses exception handling to handle cases like issuing books, returning
books, etc.
*/

import java.util.Scanner;

class Book
{
    int id;
    String name, author;
    boolean available = true;

    void addBook()
    {
        Scanner sc = new Scanner(System.in);

        System.out.println("Enter Book ID:");
        id = sc.nextInt();
        sc.nextLine();

        System.out.println("Enter Book Name:");
        name = sc.nextLine();

        System.out.println("Enter Author Name:");
        author = sc.nextLine();
    }

    void display()
    {
        System.out.println("\nBook ID: " + id);
        System.out.println("Book Name: " + name);
        System.out.println("Author Name: " + author);

        if(available)
            System.out.println("Status: Available");
        else
            System.out.println("Status: Issued");
    }
}

class Library extends Book
{
    // Polymorphism (Method Overriding)
    void display()
    {
        super.display();

        if(available)
            System.out.println("Book can be issued.");
        else
            System.out.println("Book is not available.");
    }
}

public class LibraryManagement
{
    public static void main(String args[])
    {
        Scanner sc = new Scanner(System.in);

        Library b[] = new Library[10];

        int count = 0, choice;

        do
        {
            System.out.println("\n===== LIBRARY MENU =====");

            System.out.println("1. Add Book");
            System.out.println("2. Display Books");
            System.out.println("3. Issue Book");
            System.out.println("4. Return Book");
            System.out.println("5. Exit");

            System.out.println("Enter Choice:");
            choice = sc.nextInt();

            switch(choice)
            {
                case 1:

                    b[count] = new Library();

                    b[count].addBook();

                    count++;

                    System.out.println("Book Added.");
                    break;

                case 2:

                    if(count == 0)
                    {
                        System.out.println("No Books Found.");
                    }
                    else
                    {
                        for(int i=0; i<count; i++)
                        {
                            b[i].display();
                        }
                    }

                    break;

                case 3:

                    try
                    {
                        System.out.println(
                                "Enter Book ID to Issue:");

                        int issueId = sc.nextInt();

                        boolean found = false;

                        for(int i=0; i<count; i++)
                        {
                            if(b[i].id == issueId)
                            {
                                found = true;

                                if(b[i].available)
                                {
                                    b[i].available = false;

                                    System.out.println(
                                            "Book Issued Successfully.");
                                }
                                else
                                {
                                    throw new Exception(
                                            " Exception : Book Already Issued.");
                                }
                            }
                        }

                        if(!found)
                        {
                            throw new Exception(
                                    "Invalid Book ID.");
                        }
                    }

                    catch(Exception e)
                    {
                        System.out.println(e.getMessage());
                    }

                    break;

                case 4:

                    try
                    {
                        System.out.println(
                                "Enter Book ID to Return:");

                        int returnId = sc.nextInt();

                        boolean found = false;

                        for(int i=0; i<count; i++)
                        {
                            if(b[i].id == returnId)
                            {
                                found = true;

                                if(!b[i].available)
                                {
                                    b[i].available = true;

                                    System.out.println(
                                            "Book Returned Successfully.");
                                }
                                else
                                {
                                    throw new Exception(
                                            "Exception : Book Already Returned.");
                                }
                            }
                        }

                        if(!found)
                        {
                            throw new Exception(
                                    "Invalid Book ID.");
                        }
                    }

                    catch(Exception e)
                    {
                        System.out.println(e.getMessage());
                    }

                    break;

                case 5:

                    System.out.println("Exiting...");
                    break;

                default:

                    System.out.println("Invalid Choice");
            }

        } while(choice != 5);
    }
}
